III. KẾT NỐI ỨNG DỤNG WINDOWS FORMS VỚI CSDL
3.1. CHUẨN BỊ:
- CƠ SỞ DỮ LIỆU:
VD: TẠO CSDL PTUDCSDLN0X GỒM BẢNG
LoaiHang(MaLoaiHang,TenLoaiHang)
    MaLoaiHang LÀ CHUỖI TỐI ĐA 3 KÝ TỰ, LÀ KHÓA CHÍNH
    TenLoaiHang LÀ CHUỖI TỐI ĐA 30 KÝ TỰ, KO ĐỂ TRỐNG, KO TRÙNG LẶP
DonViTinh(MaDonViTinh,TenDonViTinh,MaDonViTinhGoc,QuyDoi)
    MaDonViTinh LÀ 1 CHUỖI TỐI ĐA 3 KÝ TỰ, LÀ KHÓA CHÍNH
    TenDonViTinh LÀ CHUỖI TỐI ĐA 30 KÝ TỰ, KO ĐỂ TRỐNG, KO TRÙNG LẶP
    MaDonViTinhGoc LÀ 1 CHUỖI TỐI ĐA 3 KÝ TỰ, THAM CHIẾU SANG MaDonViTinh
    QuyDoi LÀ 1 SỐ THỰC, KO ĐỂ TRỐNG, MẶC ĐỊNH LÀ 1
TẠO DỮ LIỆU MỖI BẢNG 5 DÒNG DỮ LIỆU
CREATE DATABASE PTUDCSDLN0X
USE PTUDCSDLN0X
CREATE TABLE LoaiHang (
    MaLoaiHang VARCHAR(3) PRIMARY KEY,
    TenLoaiHang NVARCHAR(30) NOT NULL UNIQUE
)
CREATE TABLE DonViTinh (
    MaDonViTinh VARCHAR(3) PRIMARY KEY,
    TenDonViTinh NVARCHAR(30) NOT NULL UNIQUE,
    MaDonViTinhGoc VARCHAR(3) REFERENCES DonViTinh(MaDonViTinh),
    QuyDoi FLOAT NOT NULL DEFAULT(1)
)
